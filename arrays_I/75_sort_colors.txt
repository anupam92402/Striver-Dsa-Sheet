class Solution {
    public void sortColors(int[] nums) {
        int n = nums.length;
        int start = 0, end = n - 1, i = 0;
        while(i <= end){
            if(nums[i] == 0){
                nums[i] = nums[start];
                nums[start++] = 0;
            }else if(nums[i] == 2){
                nums[i] = nums[end];
                nums[end--] = 2;
                i--;
            }
            i++;
            
        }
    }
}
